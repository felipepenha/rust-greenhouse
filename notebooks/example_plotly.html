fn scatter_plot(show: bool, x: Vec<i32>, y: Vec<i32>) {    
    
    let trace = Scatter::new(x, y).mode(Mode::Markers);
    let mut plot = Plot::new();
    
    plot.add_trace(trace);
    
    let layout = Layout::new().height(200);
    let layout = Layout::new().width(800);
    
    plot.set_layout(layout);
    plot.notebook_display();
}
